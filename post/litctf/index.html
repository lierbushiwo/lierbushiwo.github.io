def recover_factors_z3(n_val, s_val):
    p = Int('p')
    q = Int('q')

    solver = Solver()
    solver.add(p * q == n_val)

        pv = model[p].as_long()
        qv = model[q].as_long()
        return min(pv, qv), max(pv, qv)
    else:
        return None
